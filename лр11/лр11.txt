using System;

namespace ConsoleApp11

    partial class Vrema
    {
        public int chas;
        public int min;
        public int sec;
        static int z;
        public int Chasy
        {
            get { if (chas <= 23 & chas >= 0) return 1; return chas; }
            set { chas = value; }
        }
        public int Minyt
        {
            get { if (min <= 59 & min >= 0) return 1; return min; }
            set { min = value; }
        }
        public int Secynd
        {
            get { if (sec <= 59 & sec >= 0) return 1; return sec; }
        }
        public Vrema(int a, int b, int c)
        {
            chas = a;
            min = b;
            sec = c;
            z++;
        }
        public void reWrite()
        {
            Vrema.Inform();
            Console.WriteLine("Время: " + chas + "." + min + "." + sec);
        }
        static void Inform()
        {
            Console.WriteLine("(" + z.ToString() + ")");
        }
        public void Categorii()
        {
            if (chas >= 22 || chas >= 0 & chas <= 6) Console.WriteLine("Сейчас ночь");
            if (chas > 6 & chas <= 10) Console.WriteLine("Сейчас утро");
            if (chas > 10 & chas <= 17) Console.WriteLine("Сейчас день");
            if (chas > 17 & chas <= 22) Console.WriteLine("Сейчас вечер");
        }
        public bool ravno()
        {
            bool b= false;
            if (chas == 23)  b=true;
            return b;
        }

        public void Zadanie(int x, int w, int v)
        {

            Console.WriteLine("Введите часы ");
            x = Convert.ToInt32(Console.ReadLine());
            Console.WriteLine("Введите минуты ");
            w = Convert.ToInt32(Console.ReadLine());
            Console.WriteLine("Введите секунды ");
            v = Convert.ToInt32(Console.ReadLine());
            Vrema n = new Vrema(x, w, v);
            z--;
            n.reWrite();
            if (n.Chasy == 1 & n.Minyt == 1 & n.Secynd == 1) { Console.WriteLine("Время задано корректно"); n.Categorii(); } else Console.WriteLine("Время задано не корректно");
            Console.ReadLine();
        }
    }
